\doxysubsection{Detailed Description}
\begin{DoxyVerb} This class is a 'form', it inherits class FlaskForm and provides a structure for form which accepts all the details regarding a post\end{DoxyVerb}


\begin{DoxyVerb} BooleanField, which is a checkbox to get whether the respective amenity is provided or not \end{DoxyVerb}


\begin{DoxyVerb} variable landloard_nature stores input text as string, that is, nature of landlord according to the user \end{DoxyVerb}


\begin{DoxyVerb} variable neighbours_nature stores input text as string, that is, nature of neighbours according to the user \end{DoxyVerb}


\begin{DoxyVerb} variable comments stores input text as string, that is, additional comment regarding the place \end{DoxyVerb}


\begin{DoxyVerb} variable nearby stores input text as string, that is, nearby areas \end{DoxyVerb}


\begin{DoxyVerb} variable flat_rating stores overall rating of the place as an integer from the point of user who is posting  \end{DoxyVerb}


\begin{DoxyVerb} Filefield which takes file as input of the given formats \end{DoxyVerb}


\begin{DoxyVerb} SubmitField to submit the form that has all the details regarding a post \end{DoxyVerb}
